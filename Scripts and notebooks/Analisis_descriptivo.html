def clean_tweets(twt):
    
    # Limpieza de tweets.
    # Se conservan los hashtags que mencionen el activo y los RT
    # Se borran el resto de hashtags y todas las menciones de usuarios
    # Se eliminan los saltos de linea y los hipervinculos
    
    twt = re.sub('#bitcoin', 'bitcoin', twt) 
    twt = re.sub('#Bitcoin', 'bitcoin', twt)
    twt = re.sub('RT', '', twt)
    twt = re.sub('#[A-Za-z0-9-^\w]+', '', twt)
    twt = re.sub('@[A-Za-z0-9]+', '', twt)
    twt = re.sub('\\n', '', twt)
    twt = re.sub('https?:\/\/\S+', '', twt)
    twt = re.sub('[()!?:]', ' ', twt)
    twt = twt.lower()
    return twt


df["Tweets_cleaned"] = df["Tweets"].progress_apply(clean_tweets)

def get_subjectivity(twt):
    return TextBlob(twt).sentiment.subjectivity

def get_polarity(twt):
    return TextBlob(twt).sentiment.polarity

def get_sentiment(score):
    if score<0:
        return "Negative"
    elif score == 0:
        return "Neutral"
    else:
        return "Positive"

#Se incorporan las columnas con la evaluacion de sentimientos al dataset original

df["Subjectivity"] = df["Tweets_cleaned"].progress_apply(get_subjectivity)
df["Polarity"] = df["Tweets_cleaned"].progress_apply(get_polarity)
df["Sentiment"] = df["Polarity"].progress_apply(get_sentiment)
